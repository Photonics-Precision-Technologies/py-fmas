2.4. How to use a solver on its own¶

This examples shows how a propagation algorithm can be usded outside the accomanying algorithms and data-sturcutres implemented in py-fmas.

The below example shows how to perform a simulation in terms of the standard nonlinear Schrödiger equation by using the symmetric split-step Fourier method implemented in py-fmas.

r"""
How to use a solver on its own
==============================

This examples shows how a propagation algorithm can be usded outside the
accomanying algorithms and data-sturcutres implemented in `py-fmas`.

The below example shows how to perform a simulation in terms of the standard
nonlinear Schrödiger equation by using the symmetric split-step Fourier method
implemented in `py-fmas`.

.. codeauthor:: Oliver Melchert <melchert@iqo.uni-hannover.de>
"""
import sys; sys.path.append('../')
import numpy as np
import numpy.fft as nfft
from fmas.solver import SySSM

# -- DEFINITION OF FFT OPERATIONS USED BY PY-FMAS
FT = nfft.ifft
IFT = nfft.fft
FTFREQ = nfft.fftfreq

def main():

    # -- DEFINE SIMULATION PARAMETERS
    # ... WAVEGUIDE PROPERTIES
    b2 = -1.0
    gamma = 1.
    # ... TEST PULSE PROPERTIES
    t0 = 1.                       # duration
    P0 = np.abs(b2)/t0/t0/gamma   # peak-intensity
    LD = t0*t0/np.abs(b2)         # dispersion length
    # ... EXACT FUNDAMENTAL SOLITON SOLUTION
    u_exact = lambda z, t: np.sqrt(P0)*np.exp(0.5j*gamma*P0*z)/np.cosh(t/t0)
    # ... COMPUTATIONAL DOMAIN 
    t_max = 50.
    t_num = 2**12
    z_max  = 0.5*np.pi*LD
    z_num  = 1000
    z_skip = 10

    # -- INITIALIZATION STAGE
    # ... COMPUTATIONAL DOMAIN
    t = np.linspace(-t_max, t_max, t_num, endpoint=False)
    w = FTFREQ(t.size, d=t[1]-t[0])*2*np.pi
    # ... INITIAL CONDITION FOR PROPAGATION
    u0_t = u_exact(0.0, t)

    # -- FREQUENCY DOMAIN REPRESENTATION OF OPERATORS
    # ... LINEAR OPERATOR 
    Lw = 0.5j*b2*w*w
    # ... NONLINEAR OPERATOR
    def Nw_func(uw):
        ut = IFT(uw)
        return 1j*gamma*FT(np.abs(ut)**2*ut)

    # BEGIN - PY-FMAS LIBRARY CODE ############################################
    # -- Z-PROPAGATION USING SYMMETRIC SPLIT-STEP FOURIER METHOD 
    solver = SySSM(Lw, Nw_func)
    solver.set_initial_condition(w, FT(u0_t))
    solver.propagate(z_range = z_max, n_steps = z_num, n_skip = z_skip)
    # END - PY-FMAS LIBRARY CODE ##############################################


if __name__=='__main__':
    main()
